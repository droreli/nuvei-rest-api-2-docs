# Create Payment

Create a new payment (Sale or Auth) using cards, digital wallets, or bank transfers.

## Request

<ParamField body="processingEntityId" type="string" required>
  Your merchant processing entity ID
</ParamField>

<ParamField body="transactionType" type="string" required>
  Transaction type: `Sale` or `Auth`
</ParamField>

<ParamField body="amount" type="number" required>
  Amount in minor units (e.g., 100 = $1.00)
</ParamField>

<ParamField body="currency" type="string" required>
  ISO 4217 currency code (e.g., `USD`, `EUR`)
</ParamField>

<ParamField body="paymentMethod" type="object" required>
  Payment method details

  <Expandable title="paymentMethod properties">
    <ParamField body="type" type="string" required>
      Payment type: `card`, `applePay`, `googlePay`, `ach`, `paymentToken`
    </ParamField>

    <ParamField body="card" type="object">
      Card details (when type is `card`)
      
      <Expandable title="card properties">
        <ParamField body="cardNumber" type="string" required>
          Full card number
        </ParamField>
        <ParamField body="expirationMonth" type="string" required>
          Expiration month (MM)
        </ParamField>
        <ParamField body="expirationYear" type="string" required>
          Expiration year (YYYY)
        </ParamField>
        <ParamField body="cvv" type="string" required>
          Card CVV/CVC
        </ParamField>
        <ParamField body="cardHolderName" type="string">
          Cardholder name
        </ParamField>
      </Expandable>
    </ParamField>

    <ParamField body="threeD" type="object">
      3D Secure settings
      
      <Expandable title="threeD properties">
        <ParamField body="challengePreference" type="string">
          `noPreference`, `challengeRequested`, `challengeMandated`
        </ParamField>
      </Expandable>
    </ParamField>

    <ParamField body="createPaymentToken" type="boolean">
      Save card for future use
    </ParamField>

    <ParamField body="paymentTokenId" type="string">
      Token ID (when type is `paymentToken`)
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="buyerDetails" type="object">
  Customer information

  <Expandable title="buyerDetails properties">
    <ParamField body="buyerId" type="string">
      Your customer ID
    </ParamField>
    <ParamField body="email" type="string">
      Customer email
    </ParamField>
    <ParamField body="firstName" type="string">
      First name
    </ParamField>
    <ParamField body="lastName" type="string">
      Last name
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="merchantDetails" type="object">
  Merchant-specific data

  <Expandable title="merchantDetails properties">
    <ParamField body="merchantTransactionId" type="string">
      Your order/transaction ID
    </ParamField>
    <ParamField body="customData" type="string">
      Custom metadata
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="urlDetails" type="object">
  Callback URLs

  <Expandable title="urlDetails properties">
    <ParamField body="notificationUrl" type="string">
      Webhook URL for payment events
    </ParamField>
  </Expandable>
</ParamField>

## Example Request

<CodeGroup>
```bash cURL
curl -X POST https://api-sandbox.nuvei.com/payment-api/payments \
  -H "Content-Type: application/json" \
  -H "x-api-key: YOUR_API_KEY" \
  -d '{
    "processingEntityId": "1234567890",
    "transactionType": "Sale",
    "amount": 100,
    "currency": "USD",
    "paymentMethod": {
      "type": "card",
      "card": {
        "cardNumber": "4000027891380961",
        "expirationMonth": "12",
        "expirationYear": "2030",
        "cvv": "217",
        "cardHolderName": "John Smith"
      }
    },
    "merchantDetails": {
      "merchantTransactionId": "order_12345"
    }
  }'
```

```javascript JavaScript
const response = await fetch('https://api-sandbox.nuvei.com/payment-api/payments', {
  method: 'POST',
  headers: {
    'Content-Type': 'application/json',
    'x-api-key': 'YOUR_API_KEY'
  },
  body: JSON.stringify({
    processingEntityId: '1234567890',
    transactionType: 'Sale',
    amount: 100,
    currency: 'USD',
    paymentMethod: {
      type: 'card',
      card: {
        cardNumber: '4000027891380961',
        expirationMonth: '12',
        expirationYear: '2030',
        cvv: '217',
        cardHolderName: 'John Smith'
      }
    }
  })
});
```
</CodeGroup>

## Response

<ResponseField name="paymentId" type="string">
  Unique payment identifier
</ResponseField>

<ResponseField name="transactionId" type="string">
  Nuvei transaction ID
</ResponseField>

<ResponseField name="externalTransactionId" type="string">
  External reference ID
</ResponseField>

<ResponseField name="amount" type="number">
  Transaction amount
</ResponseField>

<ResponseField name="currency" type="string">
  Currency code
</ResponseField>

<ResponseField name="transactionType" type="string">
  Type of transaction
</ResponseField>

<ResponseField name="result" type="object">
  Transaction result
  
  <Expandable title="result properties">
    <ResponseField name="status" type="string">
      `approved`, `declined`, `pending`, `error`
    </ResponseField>
    <ResponseField name="reason" type="object">
      Decline reason (if declined)
    </ResponseField>
  </Expandable>
</ResponseField>

<ResponseField name="authCode" type="string">
  Authorization code (if approved)
</ResponseField>

<ResponseField name="paymentMethod" type="object">
  Payment method details
</ResponseField>

<ResponseField name="threeD" type="object">
  3DS data (if 3DS required)
</ResponseField>
